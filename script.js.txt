console.log("const fixed:", fixed);

//Block scope vs Function scope
console.log("\nSection 4: Block vs Function Scope");

// Block scope with let/const
{
  let blockLet = "I exist only inside this block";
  const blockConst = "Me too, block-scoped";
  console.log("Inside block:", blockLet, "|", blockConst);
}
// console.log(blockLet);   // ReferenceError if uncommented
// console.log(blockConst); // ReferenceError if uncommented

// var is function-scoped (or global-scoped if not in a function)
{
  var blockVar = "I leak out of the block (function or global scope)";
}
console.log("Outside block, var survives:", blockVar);

// Function scope demo
function scopeDemo() {
  var functionVar = "I exist only inside this function (var)";
  let functionLet = "I exist only inside this function (let)";
  const functionConst = "I exist only inside this function (const)";
  console.log("Inside function:", functionVar, "|", functionLet, "|", functionConst);
}
scopeDemo();
